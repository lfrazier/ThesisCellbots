00113                 public void onSensorChanged(SensorEvent se) {
00114                         // We only care about the sensor events if the button is being
00115                         // pressed. Want to only read the value once during this event as
00116                         // mPress.pressed can change asynchronously and what we read now may
00117                         // be different
00118                         // than what we read later.
00119                         boolean pressed = mPress.pressed.get();
00120                         if (pressed) {
00121                                 // Now see if this is a transition to a new pressed state or
00122                                 // if it is part of a held pressed state.
00123                                 if (mWasPressed) {
00124                                         // Continuing a press...
00125                                         // Put in the MAX_SAMPLE_RATE_MS to allow others to get
00126                                         // processing time and as
00127                                         // a simple form of event filtering. This may not make sense
00128                                         // for the android. If
00129                                         // this app is the front app, let it use 100% of the cpu
00130                                         // reading sensors if it wants?
00131                                         long currentTimestampMs = se.timestamp / 1000000;
00132                                         long deltaTimestampMs = currentTimestampMs
00133                                                         - mLastTimestampMs;
00134                                         if (deltaTimestampMs > MAX_SAMPLE_RATE_MS) {
00135                                                 // When there is a change...
00136                                                 AccelerometerTranslator.SensorData s = new AccelerometerTranslator.SensorData(
00137                                                                 currentTimestampMs, se.values[0], se.values[1],
00138                                                                 se.values[2]);
00139                                                 mAcceleratorTranslator.movement(s);
00140                                         }
00141                                         mLastTimestampMs = currentTimestampMs;
00142                                 } else {
00143                                         // Transition from not-pressed to pressed.
00144                                         // Capture the values to use as a reference.
00145                                         AccelerometerTranslator.SensorData s = new AccelerometerTranslator.SensorData(
00146                                                         se.timestamp / 1000000, se.values[0], se.values[1],
00147                                                         se.values[2]);
00148                                         mAcceleratorTranslator.setBaseline(s);
00149                                 }
00150                         } else {
00151                                 // transition from pressed to not-pressed
00152                                 if (mWasPressed) {
00153                                         mAcceleratorTranslator.stop();
00154                                 }
00155                         }
00156                         mWasPressed = pressed;
00157                 }
