00136         public static String[] splitNameMacPair(String nameMac) {
00137                 String[] ret = new String[2];
00138 
00139                 Matcher m = NAME_MAC_SPLITTER.matcher(nameMac);
00140                 if (m.matches()) {
00141                         if (m.groupCount() == 2) {
00142                                 ret[0] = m.group(1);
00143                                 if (BluetoothAdapter.checkBluetoothAddress(m.group(2))) {
00144                                         ret[1] = m.group(2);
00145                                 } else {
00146                                         Log.w(TAG, "Problem splitting - bad mac address: "
00147                                                         + nameMac);
00148                                         ret[1] = "";
00149                                 }
00150                         } else {
00151                                 Log.e(TAG, "Problem splitting - count wrong: " + nameMac + " ("
00152                                                 + m.groupCount() + ")");
00153                                 ret[0] = nameMac;
00154                                 ret[1] = "";
00155                         }
00156                 } else {
00157                         Log.w(TAG, "Problem splitting - no match: " + nameMac);
00158                         ret[0] = nameMac;
00159                         ret[1] = "";
00160                 }
00161 
00162                 return ret;
