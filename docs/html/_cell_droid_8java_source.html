00067         public static String stripTimestampIfAny(String message) {
00068                 // TODO (chaitanyag): Change this after we come up with a better
00069                 // protocol. The first (space separated) token in the command
00070                 // string could be a timestamp. This is useful if the same commands
00071                 // are sent back to back. For example, the controller sends
00072                 // consecutive "hu" (head up) commands to tilt the head up in small
00073                 // increments.
00074                 if (message.indexOf(' ') >= 0) {
00075                         try {
00076                                 Long.parseLong(message.substring(0, message.indexOf(' ')));
00077                                 message = message.substring(message.indexOf(' ') + 1);
00078                         } catch (NumberFormatException e) {
00079                         }
00080                 }
00081                 return message;
00082         }
