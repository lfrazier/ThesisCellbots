00355                         mListener.onActionRequested(UiEventListener.ACTION_BACKWARD, null);
00356                 } else if (CMD_LEFT.equals(command)) {
00357                         mListener.onActionRequested(UiEventListener.ACTION_LEFT, null);
00358                 } else if (CMD_RIGHT.equals(command)) {
00359                         mListener.onActionRequested(UiEventListener.ACTION_RIGHT, null);
00360                 } else if (CMD_PICTURE.equals(command)) {
00361                         mListener.onActionRequested(UiEventListener.ACTION_TAKE_PICTURE,
00362                                         null);
00363                 } else if (CMD_LOCATION.equals(command)) {
00364                         mListener.onActionRequested(UiEventListener.ACTION_GET_LOCATION,
00365                                         null);
00366                 } else if (CMD_SPEAK.equals(command)) {
00367                         mListener.onActionRequested(UiEventListener.ACTION_SPEAK, "");
00368                         return;

00370                         mListener.onStopRequested();
